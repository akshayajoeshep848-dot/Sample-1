             // print numbers from 0 to 1

        //for odd numbers upto 10
        for (let i = 1; i <= 10; i = i + 2) {
            console.log(i)
        }

        }


        //sum of numbers from 1 t 10

        //To print tables upto 10
        let num = prompt("Enter number")
        for (let i = 1; i <= 10; i = i + 1) {
            console.log(num, "*", i, "=", num * i)
        }


        // To print numbers in reverse
        for (let i = 10; i >= 1; i = i - 1) {
            console.log(i)
        }

        // To print number of vowels in give word
        let str = prompt("Enter string: ");
        let count = 0;
        let vowels = "aeiou";
        for (let i = 0; i < str.length; i++) {
            if (vowels.includes(str[i].toLowerCase())) {
                count++;
            }
        }
        console.log(count);
