Switch case exercise

    write switch case for
  • grade A - 100 - 90
  • grade B - 90 - 80
  • grade C - 80 - 70
  • grade D - 70 - 60
  • grade E - 60 - 40
  • grade F - 40 - 0
        
        
Code to determine grade based on marks

    let marks = prompt("Please enter your marks")
        switch (true) {                                 
            case (marks < 0 || marks > 100):
                console.log("Invalid marks");
                break;
            case (marks >= 0 && marks < 41):
                console.log("F");
                break;
            case (marks >= 41 && marks < 61):
                console.log("E");
                break;
            case (marks >= 61 && marks < 70):
                console.log("D");
                break;
            case (marks >= 71 && marks < 81):
                console.log("C");
                break;
            case (marks >= 81 && marks < 91):
                console.log("B");
                break;
            case (marks >= 91 && marks <= 100):
                console.log("A");
                break;
        }
