Ternary

        
            
1. Even / Odd

            let num = prompt("Enter the number")
            num % 2 == 0 ? console.log("Even number") : console.log("Odd number")
            
2. Positive / Negative

            let num = prompt("Enter number")
            num < 0 ? console.log("Negative") : num > 0 ? console.log("Positive") : console.log("Zero")
            
3. Log in status

            let isLoggedIn = "true";
            let status = prompt("Log in status (true / flase): ")
            status == isLoggedIn ? console.log("Logged in") : console.log("Logged out")                 
            
4. Greatest of two numbers

            let a = prompt("Enter a number");
            let b = prompt("Enter another number");
            a > b ? console.log(a) : console.log(b);
            
5. Vote eligibility

            let age = prompt("Please enter age: ")
            age < 0 || age > 128 ? console.log("Invalid age") : age >= 18 ? console.log("Elligible") : console.log("Not elligible");
