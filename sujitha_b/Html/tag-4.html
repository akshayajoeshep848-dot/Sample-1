     def fibonacci_iterative(n):
    """
    Generates the Fibonacci sequence up to n terms using an iterative approach.

    Args:
        n (int): The number of terms to generate in the Fibonacci sequence.

    Returns:
        list: A list containing the Fibonacci sequence up to n terms.
              Returns an empty list if n is less than or equal to 0.
    """
    if n <= 0:
        return []
    elif n == 1:
        return [0]
    else:
        fib_sequence = [0, 1]
        while len(fib_sequence) < n:
            next_fib = fib_sequence[-1] + fib_sequence[-2]
            fib_sequence.append(next_fib)
        return fib_sequence
